243 /* This function allows for duplicate names in the table of named substrings.
244 It returns the number of the first one that was set in a pattern match.
245 
246 Arguments:
247  code the compiled regex
248  stringname the name of the capturing substring
249  ovector the vector of matched substrings
250

252  or the number of the last one if none are set,
253  or a negative number on error
254 */

259 #elif defined COMPILE_PCRE16

262 #elif defined COMPILE_PCRE32

264 get_first_set(const pcre32 *code, PCRE_SPTR32 stringname, int *ovector)
265 #endif
266 {
267 const REAL_PCRE *re = (const REAL_PCRE *)code;
268 int entrysize;
269 pcre_uchar *entry;
270 #if defined COMPILE_PCRE8
271 char *first, *last;
272 #elif defined COMPILE_PCRE16
273 PCRE_UCHAR16 *first, *last;
274 #elif defined COMPILE_PCRE32
275 PCRE_UCHAR32 *first, *last;
276 #endif
277 
278 #if defined COMPILE_PCRE8
279 if ((re->options & PCRE_DUPNAMES) == 0 && (re->flags & PCRE_JCHANGED) == 0)
280  return pcre_get_stringnumber(code, stringname);
281 entrysize = pcre_get_stringtable_entries(code, stringname, &first, &last);
282 #elif defined COMPILE_PCRE16
283 if ((re->options & PCRE_DUPNAMES) == 0 && (re->flags & PCRE_JCHANGED) == 0)
284  return pcre16_get_stringnumber(code, stringname);
285 entrysize = pcre16_get_stringtable_entries(code, stringname, &first, &last);
286 #elif defined COMPILE_PCRE32
287 if ((re->options & PCRE_DUPNAMES) == 0 && (re->flags & PCRE_JCHANGED) == 0)
288  return pcre32_get_stringnumber(code, stringname);
289 entrysize = pcre32_get_stringtable_entries(code, stringname, &first, &last);
290 #endif
291 if (entrysize <= 0) return entrysize;
292 for (entry = (pcre_uchar *)first; entry <= (pcre_uchar *)last; entry += entrysize)

303 /*************************************************
304 * Copy captured string to given buffer *
305 *************************************************/
306 
307 /* This function copies a single captured substring into a given buffer.
308 Note that we use memcpy() rather than strncpy() in case there are binary zeros
309 in the string.
310 
311 Arguments:
312  subject the subject string that was matched
313  ovector pointer to the offsets table
314  stringcount the number of substrings that were captured
315  (i.e. the yield of the pcre_exec call, unless
316  that was zero, in which case it should be 1/3
317  of the offset table size)
318  stringnumber the number of the required substring
319  buffer where to put the substring
320  size the size of the buffer
321 
322 Returns: if successful:
323  the length of the copied string, not including the zero
324  that is put on the end; can be zero
325  if not successful:
326  PCRE_ERROR_NOMEMORY (-6) buffer too small
327  PCRE_ERROR_NOSUBSTRING (-7) no such captured substring
328 */
329 
330 #if defined COMPILE_PCRE8
331 PCRE_EXP_DEFN int PCRE_CALL_CONVENTION
332 pcre_copy_substring(const char *subject, int *ovector, int stringcount,
333  int stringnumber, char *buffer, int size)
334 #elif defined COMPILE_PCRE16
335 PCRE_EXP_DEFN int PCRE_CALL_CONVENTION
336 pcre16_copy_substring(PCRE_SPTR16 subject, int *ovector, int stringcount,
337  int stringnumber, PCRE_UCHAR16 *buffer, int size)
338 #elif defined COMPILE_PCRE32
339 PCRE_EXP_DEFN int PCRE_CALL_CONVENTION
340 pcre32_copy_substring(PCRE_SPTR32 subject, int *ovector, int stringcount,
341  int stringnumber, PCRE_UCHAR32 *buffer, int size)
342 #endif
343 {
344 int yield;
345 if (stringnumber < 0 || stringnumber >= stringcount)
346  return PCRE_ERROR_NOSUBSTRING;
347 stringnumber *= 2;
348 yield = ovector[stringnumber+1] - ovector[stringnumber];
349 if (size < yield + 1) return PCRE_ERROR_NOMEMORY;
350 memcpy(buffer, subject + ovector[stringnumber], IN_UCHARS(yield));
351 buffer[yield] = 0;
352 return yield;
353 }

438 pcre_get_substring_list(const char *subject, int *ovector, int stringcount,
439  const char ***listptr)
440 #elif defined COMPILE_PCRE16
441 PCRE_EXP_DEFN int PCRE_CALL_CONVENTION
442 pcre16_get_substring_list(PCRE_SPTR16 subject, int *ovector, int stringcount,
443  PCRE_SPTR16 **listptr)
444 #elif defined COMPILE_PCRE32
445 PCRE_EXP_DEFN int PCRE_CALL_CONVENTION
446 pcre32_get_substring_list(PCRE_SPTR32 subject, int *ovector, int stringcount,
447  PCRE_SPTR32 **listptr)
448 #endif
449 {
450 int i;
451 int size = sizeof(pcre_uchar *);
452 int double_count = stringcount * 2;
453 pcre_uchar **stringlist;
454 pcre_uchar *p;
455 
456 for (i = 0; i < double_count; i += 2)
457  size += sizeof(pcre_uchar *) + IN_UCHARS(ovector[i+1] - ovector[i] + 1);
458 
459 stringlist = (pcre_uchar **)(PUBL(malloc))(size);
460 if (stringlist == NULL) return PCRE_ERROR_NOMEMORY;
461 
462 #if defined COMPILE_PCRE8
463 *listptr = (const char **)stringlist;
464 #elif defined COMPILE_PCRE16
465 *listptr = (PCRE_SPTR16 *)stringlist;
466 #elif defined COMPILE_PCRE32
467 *listptr = (PCRE_SPTR32 *)stringlist;
468 #endif
469 p = (pcre_uchar *)(stringlist + stringcount + 1);
470 
471 for (i = 0; i < double_count; i += 2)
472  {
473  int len = ovector[i+1] - ovector[i];
474  memcpy(p, subject + ovector[i], IN_UCHARS(len));
475  *stringlist++ = p;
476  p += len;
477  *p++ = 0;
478  }
479 
480 *stringlist = NULL;
481 return 0;
482 }
