120 inline UInt16 ByteOrder::flipBytes(UInt16 value)
121 {
122 #if defined(POCO_HAVE_MSC_BYTESWAP)
123  return _byteswap_ushort(value);
124 #else
125  return ((value >> 8) & 0x00FF) | ((value << 8) & 0xFF00);
126 #endif
127 }

136 inline UInt32 ByteOrder::flipBytes(UInt32 value)
137 {
138 #if defined(POCO_HAVE_MSC_BYTESWAP)
139  return _byteswap_ulong(value);
140 #elif defined(POCO_HAVE_GCC_BYTESWAP)
141  return __builtin_bswap32(value);
142 #else
143  return ((value >> 24) & 0x000000FF) | ((value >> 8) & 0x0000FF00)
144  | ((value << 8) & 0x00FF0000) | ((value << 24) & 0xFF000000);
145 #endif
146 }

156 inline UInt64 ByteOrder::flipBytes(UInt64 value)
157 {
158 #if defined(POCO_HAVE_MSC_BYTESWAP)
159  return _byteswap_uint64(value);
160 #elif defined(POCO_HAVE_GCC_BYTESWAP)
161  return __builtin_bswap64(value);
162 #else
163  UInt32 hi = UInt32(value >> 32);
164  UInt32 lo = UInt32(value & 0xFFFFFFFF);
165  return UInt64(flipBytes(hi)) | (UInt64(flipBytes(lo)) << 32);
