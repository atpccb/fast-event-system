45 // Double operations detection based on target architecture.
46 // Linux uses a 80bit wide floating point stack on x86. This induces double
47 // rounding, which in turn leads to wrong results.
48 // An easy way to test if the floating-point operations are correct is to
49 // evaluate: 89255.0/1e22. If the floating-point stack is 64 bits wide then
50 // the result is equal to 89255e-22.
51 // The best way to test this, is to create a division-function and to compare
52 // the output of the division with the expected result. (Inlining must be
53 // disabled.)
54 // On Linux,x86 89255e-22 != Div_double(89255.0/1e22)
55 #if defined(_M_X64) || defined(__x86_64__) || \
56  defined(__ARMEL__) || defined(_M_ARM) || defined(__arm__) || defined(__arm64__) || \
57  defined(__avr32__) || \
58  defined(__hppa__) || defined(__ia64__) || \
59  defined(__mips__) || defined(__powerpc__) || \
60  defined(__sparc__) || defined(__sparc) || defined(__s390__) || \
61  defined(__SH4__) || defined(__alpha__) || \
62  defined(_MIPS_ARCH_MIPS32R2) || \
63  defined(__AARCH64EL__) || \
64  defined(nios2) || defined(__nios2) || defined(__nios2__)
65 #define DOUBLE_CONVERSION_CORRECT_DOUBLE_OPERATIONS 1
66 #elif defined(_M_IX86) || defined(__i386__) || defined(__i386)
67 #if defined(_WIN32)
68 // Windows uses a 64bit wide floating point stack.
69 #define DOUBLE_CONVERSION_CORRECT_DOUBLE_OPERATIONS 1
70 #else
71 #undef DOUBLE_CONVERSION_CORRECT_DOUBLE_OPERATIONS
72 #endif // _WIN32
73 #else
74 #error Target architecture was not detected as supported by Double-Conversion.
75 #endif

165  // Returns a vector using the same backing storage as this one,
166  // spanning from and including 'from', to but not including 'to'.
167  Vector<T> SubVector(int from, int to) {
168  ASSERT(to <= length_);

199 // Helper class for building result strings in a character buffer. The
200 // purpose of the class is to use safe operations that checks the
201 // buffer bounds on all operations in debug mode.
202 class StringBuilder {
203  public:
204  StringBuilder(char* buffer, int size)
205  : buffer_(buffer, size), position_(0) { }
206 
207  ~StringBuilder() { if (!is_finalized()) Finalize(); }
208 
209  int size() const { return buffer_.length(); }
210 
211  // Get the current position in the builder.
212  int position() const {
213  ASSERT(!is_finalized());
214  return position_;
215  }
216 
217  // Reset the position.
218  void Reset() { position_ = 0; }
219 
220  // Add a single character to the builder. It is not allowed to add
221  // 0-characters; use the Finalize() method to terminate the string
222  // instead.
223  void AddCharacter(char c) {
224  ASSERT(c != '\0');
225  ASSERT(!is_finalized() && position_ < buffer_.length());
226  buffer_[position_++] = c;
227  }
228 
229  // Add an entire string to the builder. Uses strlen() internally to
230  // compute the length of the input string.
231  void AddString(const char* s) {
232  AddSubstring(s, StrLength(s));
233  }
234 
235  // Add the first 'n' characters of the given string 's' to the
236  // builder. The input string must have enough characters.
237  void AddSubstring(const char* s, int n) {
238  ASSERT(!is_finalized() && position_ + n < buffer_.length());
239  ASSERT(static_cast<size_t>(n) <= strlen(s));
240  memmove(&buffer_[position_], s, n * kCharSize);
241  position_ += n;
242  }
243 
244 
245  // Add character padding to the builder. If count is non-positive,
246  // nothing is added to the builder.
247  void AddPadding(char c, int count) {
248  for (int i = 0; i < count; i++) {
249  AddCharacter(c);
250  }
251  }
252 
253  // Finalize the string by 0-terminating it and returning the buffer.
254  char* Finalize() {
255  ASSERT(!is_finalized() && position_ < buffer_.length());
256  buffer_[position_] = '\0';
257  // Make sure nobody managed to add a 0-character to the
258  // buffer while building the string.
259  ASSERT(strlen(buffer_.start()) == static_cast<size_t>(position_));
260  position_ = -1;
261  ASSERT(is_finalized());
262  return buffer_.start();
263  }
264 
265  private:
266  Vector<char> buffer_;
267  int position_;
268 
269  bool is_finalized() const { return position_ < 0; }
270 
271  DISALLOW_IMPLICIT_CONSTRUCTORS(StringBuilder);
272 };

274 // The type-based aliasing rule allows the compiler to assume that pointers of
275 // different types (for some definition of different) never alias each other.
276 // Thus the following code does not work:
277 //
278 // float f = foo();
279 // int fbits = *(int*)(&f);
280 //
281 // The compiler 'knows' that the int pointer can't refer to f since the types
282 // don't match, so the compiler may cache f in a register, leaving random data
283 // in fbits. Using C++ style casts makes no difference, however a pointer to
284 // char data is assumed to alias any other pointer. This is the 'memcpy
285 // exception'.
286 //
287 // Bit_cast uses the memcpy exception to move the bits from a variable of one
288 // type of a variable of another type. Of course the end result is likely to
289 // be implementation dependent. Most compilers (gcc-4.2 and MSVC 2005)
290 // will completely optimize BitCast away.
291 //
292 // There is an additional use for BitCast.
293 // Recent gccs will warn when they see casts that may result in breakage due to
294 // the type-based aliasing rule. If you have checked that there is no breakage
295 // you can use BitCast to cast one pointer type to another. This confuses gcc
296 // enough that it can no longer see that you have cast one pointer type to
297 // another thus avoiding the warning.
298 template <class Dest, class Source>
299 inline Dest BitCast(const Source& source) {
300  // Compile time assertion: sizeof(Dest) == sizeof(Source)
301  // A compile error here means your Dest and Source have different sizes.
302  typedef char VerifySizesAreEqual[sizeof(Dest) == sizeof(Source) ? 1 : -1];
303 
304  Dest dest;
305  memmove(&dest, &source, sizeof(dest));
306  return dest;
307 }
308 
309 template <class Dest, class Source>
310 inline Dest BitCast(Source* source) {
311  return BitCast<Dest>(reinterpret_cast<uintptr_t>(source));
312 }
