116 int UTF8Encoding::convert(int ch, unsigned char* bytes, int length) const
117 {
118  if (ch <= 0x7F)
119  {
120  if (bytes && length >= 1)
121  *bytes = (unsigned char) ch;
122  return 1;
123  }
124  else if (ch <= 0x7FF)
125  {
126  if (bytes && length >= 2)
127  {
128  *bytes++ = (unsigned char) (((ch >> 6) & 0x1F) | 0xC0);
129  *bytes = (unsigned char) ((ch & 0x3F) | 0x80);
130  }
131  return 2;
132  }
133  else if (ch <= 0xFFFF)
134  {
135  if (bytes && length >= 3)
136  {
137  *bytes++ = (unsigned char) (((ch >> 12) & 0x0F) | 0xE0);
138  *bytes++ = (unsigned char) (((ch >> 6) & 0x3F) | 0x80);
139  *bytes = (unsigned char) ((ch & 0x3F) | 0x80);
140  }
141  return 3;
142  }
143  else if (ch <= 0x10FFFF)
144  {
145  if (bytes && length >= 4)
146  {

207 bool UTF8Encoding::isLegal(const unsigned char *bytes, int length)
208 {
209  // Note: The following is loosely based on the isLegalUTF8 function
210  // from ftp://ftp.unicode.org/Public/PROGRAMS/CVTUTF/ConvertUTF.c
211  // Excuse the ugliness...
212 
213  if (0 == bytes || 0 == length) return false;
214 
215  unsigned char a;
216  const unsigned char* srcptr = bytes + length;
217  switch (length)
218  {
219  default:
220  return false;
221  // Everything else falls through when true.
222  case 4:
223  if ((a = (*--srcptr)) < 0x80 || a > 0xBF) return false;
224  case 3:
225  if ((a = (*--srcptr)) < 0x80 || a > 0xBF) return false;
226  case 2:
227  if ((a = (*--srcptr)) > 0xBF) return false;
228  switch (*bytes)
229  {
230  case 0xE0:
231  if (a < 0xA0) return false;
232  break;
233  case 0xED:
234  if (a > 0x9F) return false;
235  break;
236  case 0xF0:
237  if (a < 0x90) return false;
238  break;
239  case 0xF4:
240  if (a > 0x8F) return false;
241  break;
242  default:
243  if (a < 0x80) return false;
244  }
245  case 1:
246  if (*bytes >= 0x80 && *bytes < 0xC2) return false;
247  }
248  return *bytes <= 0xF4;
249 }
