57 /*
58  * random.c:
59  *
60  * An improved random number generation package. In addition to the standard
61  * rand()/srand() like interface, this package also has a special state info
62  * interface. The initstate() routine is called with a seed, an array of
63  * bytes, and a count of how many bytes are being passed in; this array is
64  * then initialized to contain information for random number generation with
65  * that much state information. Good sizes for the amount of state
66  * information are 32, 64, 128, and 256 bytes. The state can be switched by
67  * calling the setstate() routine with the same array as was initiallized
68  * with initstate(). By default, the package runs with 128 bytes of state
69  * information and generates far better random numbers than a linear
70  * congruential generator. If the amount of state information is less than
71  * 32 bytes, a simple linear congruential R.N.G. is used.
72  *
73  * Internally, the state information is treated as an array of uint32_t's; the
74  * zeroeth element of the array is the type of R.N.G. being used (small
75  * integer); the remainder of the array is the state information for the
76  * R.N.G. Thus, 32 bytes of state information will give 7 ints worth of
77  * state information, which will allow a degree seven polynomial. (Note:
78  * the zeroeth word of state information also has some other information
79  * stored in it -- see setstate() for details).
80  *
81  * The random number generation technique is a linear feedback shift register
82  * approach, employing trinomials (since there are fewer terms to sum up that
83  * way). In this approach, the least significant bit of all the numbers in
84  * the state table will act as a linear feedback shift register, and will
85  * have period 2^deg - 1 (where deg is the degree of the polynomial being
86  * used, assuming that the polynomial is irreducible and primitive). The
87  * higher order bits will have longer periods, since their values are also
88  * influenced by pseudo-random carries out of the lower bits. The total
89  * period of the generator is approximately deg*(2**deg - 1); thus doubling
90  * the amount of state information has a vast influence on the period of the
91  * generator. Note: the deg*(2**deg - 1) is an approximation only good for
92  * large deg, when the period of the shift is the dominant factor.
93  * With deg equal to seven, the period is actually much longer than the
94  * 7*(2**7 - 1) predicted by this formula.
95  *
96  * Modified 28 December 1994 by Jacob S. Rosenberg.
97  * The following changes have been made:
98  * All references to the type u_int have been changed to unsigned long.
99  * All references to type int have been changed to type long. Other
100  * cleanups have been made as well. A warning for both initstate and
101  * setstate has been inserted to the effect that on Sparc platforms
102  * the 'arg_state' variable must be forced to begin on word boundaries.
103  * This can be easily done by casting a long integer array to char *.
104  * The overall logic has been left STRICTLY alone. This software was
105  * tested on both a VAX and Sun SpacsStation with exactly the same
106  * results. The new version and the original give IDENTICAL results.
107  * The new version is somewhat faster than the original. As the
108  * documentation says: "By default, the package runs with 128 bytes of
109  * state information and generates far better random numbers than a linear
110  * congruential generator. If the amount of state information is less than
111  * 32 bytes, a simple linear congruential R.N.G. is used." For a buffer of
112  * 128 bytes, this new version runs about 19 percent faster and for a 16
113  * byte buffer it is about 5 percent faster.
114  */

172 /*
173  * Compute x = (7^5 * x) mod (2^31 - 1)
174  * wihout overflowing 31 bits:
175  * (2^31 - 1) = 127773 * (7^5) + 2836
176  * From "Random number generators: good ones are hard to find",
177  * Park and Miller, Communications of the ACM, vol. 31, no. 10,
178  * October 1988, p. 1195.
179  */
180 inline UInt32 Random::goodRand(Int32 x)
181 {
182  Int32 hi, lo;
183 
184  if (x == 0) x = 123459876;
185  hi = x / 127773;
186  lo = x % 127773;
187  x = 16807 * lo - 2836 * hi;
188  if (x < 0) x += 0x7FFFFFFF;
189 
190  return x;
191 }

194 /*
195  * Initialize the random number generator based on the given seed. If the
196  * type is the trivial no-state-information type, just remember the seed.
197  * Otherwise, initializes state[] based on the given "seed" via a linear
198  * congruential generator. Then, the pointers are set to known locations
199  * that are exactly rand_sep places apart. Lastly, it cycles the state
200  * information a given number of times to get rid of any initial dependencies
201  * introduced by the L.C.R.N.G. Note that the initialization of randtbl[]
202  * for default usage relies on values produced by this routine.
203  */
204 void Random::seed(UInt32 x)
205 {
206  int i, lim;
207 
208  _state[0] = x;
209  if (_randType == TYPE_0)
210  lim = NSHUFF;
211  else
212  {
213  for (i = 1; i < _randDeg; i++)
214  _state[i] = goodRand(_state[i - 1]);
215  _fptr = &_state[_randSep];
216  _rptr = &_state[0];
217  lim = 10 * _randDeg;
218  }
219  for (i = 0; i < lim; i++)
220  next();
221 }

224 /*
225  * Many programs choose the seed value in a totally predictable manner.
226  * This often causes problems. We seed the generator using the much more
227  * secure random(4) interface. Note that this particular seeding
228  * procedure can generate states which are impossible to reproduce by
229  * calling srandom() with any value, since the succeeding terms in the
230  * state buffer are no longer derived from the LC algorithm applied to
231  * a fixed seed.
232  */
233 void Random::seed()
234 {
235  std::streamsize len;
236 
237  if (_randType == TYPE_0)
238  len = sizeof _state[0];
239  else
240  len = _randDeg * sizeof _state[0];
241 
242  RandomInputStream rstr;
243  rstr.read((char*) _state, len);
244 }

247 /*
248  * Initialize the state information in the given array of n bytes for future
249  * random number generation. Based on the number of bytes we are given, and
250  * the break values for the different R.N.G.'s, we choose the best (largest)
251  * one we can and set things up for it. srandom() is then called to
252  * initialize the state information.
253  *
254  * Note that on return from srandom(), we set state[-1] to be the type
255  * multiplexed with the current value of the rear pointer; this is so
256  * successive calls to initstate() won't lose this information and will be
257  * able to restart with setstate().
258  *
259  * Note: the first thing we do is save the current state, if any, just like
260  * setstate() so that it doesn't matter when initstate is called.
261  *
262  * Returns a pointer to the old state.
263  *
264  * Note: The Sparc platform requires that arg_state begin on an int
265  * word boundary; otherwise a bus error will occur. Even so, lint will
266  * complain about mis-alignment, but you should disregard these messages.
267  */
268 void Random::initState(UInt32 s, char* argState, Int32 n)
269 {
270  UInt32* intArgState = (UInt32*) argState;
271 
272  if (n < BREAK_0)
273  {
274  poco_bugcheck_msg("not enough state");
275  return;
276  }
277  if (n < BREAK_1)
278  {
279  _randType = TYPE_0;
280  _randDeg = DEG_0;
281  _randSep = SEP_0;
282  }
283  else if (n < BREAK_2)
284  {
285  _randType = TYPE_1;
286  _randDeg = DEG_1;
287  _randSep = SEP_1;
288  }
289  else if (n < BREAK_3)
290  {
291  _randType = TYPE_2;
292  _randDeg = DEG_2;
293  _randSep = SEP_2;
294  }
295  else if (n < BREAK_4)
296  {
297  _randType = TYPE_3;
298  _randDeg = DEG_3;
299  _randSep = SEP_3;
300  }
301  else
302  {
303  _randType = TYPE_4;
304  _randDeg = DEG_4;
305  _randSep = SEP_4;
306  }
307  _state = intArgState + 1; /* first location */
308  _endPtr = &_state[_randDeg]; /* must set end_ptr before seed */
309  seed(s);
310  if (_randType == TYPE_0)
311  intArgState[0] = _randType;
312  else
313  intArgState[0] = MAX_TYPES * (int) (_rptr - _state) + _randType;
314 }

317 /*
318  * Next:
319  *
320  * If we are using the trivial TYPE_0 R.N.G., just do the old linear
321  * congruential bit. Otherwise, we do our fancy trinomial stuff, which is
322  * the same in all the other cases due to all the global variables that have
323  * been set up. The basic operation is to add the number at the rear pointer
324  * into the one at the front pointer. Then both pointers are advanced to
325  * the next location cyclically in the table. The value returned is the sum
326  * generated, reduced to 31 bits by throwing away the "least random" low bit.
327  *
328  * Note: the code takes advantage of the fact that both the front and
329  * rear pointers can't wrap on the same call by not testing the rear
330  * pointer if the front one has wrapped.
331  *
332  * Returns a 31-bit random number.
333  */
334 UInt32 Random::next()
335 {
336  UInt32 i;
337  UInt32 *f, *r;
338 
339  if (_randType == TYPE_0)
340  {
341  i = _state[0];
342  _state[0] = i = goodRand(i) & 0x7FFFFFFF;
343  }
344  else
345  {
346  /*
347  * Use local variables rather than static variables for speed.
348  */
349  f = _fptr; r = _rptr;
350  *f += *r;
351  i = (*f >> 1) & 0x7FFFFFFF; /* chucking least random bit */
352  if (++f >= _endPtr) {
353  f = _state;
354  ++r;
355  }
356  else if (++r >= _endPtr) {
357  r = _state;
358  }
359 
360  _fptr = f; _rptr = r;
361  }
362  return i;
363 }
