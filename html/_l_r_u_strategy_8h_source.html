59  void onAdd(const void*, const KeyValueArgs <TKey, TValue>& args)
60  {
61  _keys.push_front(args.key());
62  std::pair<IndexIterator, bool> stat = _keyIndex.insert(std::make_pair(args.key(), _keys.begin()));
63  if (!stat.second)
64  {
65  stat.first->second = _keys.begin();
66  }

69  void onRemove(const void*, const TKey& key)
70  {
71  IndexIterator it = _keyIndex.find(key);
72 
73  if (it != _keyIndex.end())
74  {
75  _keys.erase(it->second);
76  _keyIndex.erase(it);
77  }
78  }

80  void onGet(const void*, const TKey& key)
81  {
82  // LRU: in case of an hit, move to begin
83  IndexIterator it = _keyIndex.find(key);
84 
85  if (it != _keyIndex.end())
86  {
87  _keys.splice(_keys.begin(), _keys, it->second); //_keys.erase(it->second)+_keys.push_front(key);
88  it->second = _keys.begin();
89  }
90  }

93  {
94  _keys.clear();

96  }

98  void onIsValid(const void*, ValidArgs<TKey>& args)
99  {

106  void onReplace(const void*, std::set<TKey>& elemsToRemove)
107  {
108  // Note: replace only informs the cache which elements
109  // it would like to remove!
110  // it does not remove them on its own!
111  std::size_t curSize = _keyIndex.size();
112 
113  if (curSize < _size)
114  {
115  return;
116  }
117 
118  std::size_t diff = curSize - _size;
119  Iterator it = --_keys.end(); //--keys can never be invoked on an empty list due to the minSize==1 requirement of LRU
120  std::size_t i = 0;
121 
122  while (i++ < diff)
123  {
124  elemsToRemove.insert(*it);
125  if (it != _keys.begin())
126  {
127  --it;
128  }
129  }
130  }
